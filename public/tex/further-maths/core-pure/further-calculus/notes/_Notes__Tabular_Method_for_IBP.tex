\begin{tcolorbox}[
  enhanced,
  colback=softblue,
  colframe=accent,
  colbacktitle=accent!50!white,
  boxrule=0.9pt,
  arc=2mm,
  title=The IBP Formula,
  fonttitle=\bfseries,
  coltitle=black
]
Integration by parts may be written as
\[
\int u\,\frac{\mathrm{d}v}{\mathrm{d}x}\,\mathrm{d}x
= uv - \int v\,\frac{\mathrm{d}u}{\mathrm{d}x}\,\mathrm{d}x
\]
In the tabular method, the left column is differentiated repeatedly, the right column is integrated repeatedly, and the answer is read from the diagonal products with alternating signs.
\end{tcolorbox}

\begin{tcolorbox}[
  enhanced,
  colback=softgreen,
  colframe=accent2,
  colbacktitle=accent2!50!white,
  boxrule=0.9pt,
  arc=2mm,
  title=The Tabular Method,
  fonttitle=\bfseries,
  coltitle=black
]
Choose
\[
 u=\text{the factor to differentiate},
 \qquad
 \frac{\mathrm{d}v}{\mathrm{d}x}=\text{the factor to integrate}
\]
Then:
\begin{enumerate}[leftmargin=*,itemsep=0.15em,topsep=0.2em]
    \item write the alternating signs \(+, -, +, -, \dots\),
    \item differentiate the \(u\)-column downward,
    \item integrate the \(\dfrac{\mathrm{d}v}{\mathrm{d}x}\)-column downward,
    \item read each arrow diagonally to form the terms,
    \item and include the final remainder integral when needed (if the Differentiate column didn't reach 0) by adding the integral of the product of the final row.
\end{enumerate}
\end{tcolorbox}

\section*{Example 1: \(\int x^2e^x\,\mathrm{d}x\)}
Take
\[
 u=x^2,
 \qquad
 \frac{\mathrm{d}v}{\mathrm{d}x}=e^x
\]
\tableA
So
\[
\int x^2e^x\,\mathrm{d}x=x^2e^x-2xe^x+2e^x
\]
Hence
\[
\boxed{\int x^2e^x\,\mathrm{d}x=e^x(x^2-2x+2)+C}
\]

\newpage
\section*{Example 2: \(\int x^3\sin x\,\mathrm{d}x\)}
Take
\[
 u=x^3,
 \qquad
 \frac{\mathrm{d}v}{\mathrm{d}x}=\sin x
\]
\tableB
Reading the diagonals gives
\[
\boxed{\int x^3\sin x\,\mathrm{d}x=-x^3\cos x+3x^2\sin x+6x\cos x-6\sin x+C}
\]

\section*{Example 3: \(\int x^4e^{2x}\,\mathrm{d}x\)}
Take
\[
 u=x^4,
 \qquad
 \frac{\mathrm{d}v}{\mathrm{d}x}=e^{2x}
\]
\tableC
So
\[
\int x^4e^{2x}\,\mathrm{d}x
=\frac12x^4e^{2x}-x^3e^{2x}+\frac32x^2e^{2x}-\frac32xe^{2x}+\frac34e^{2x}+C
\]
Hence
\[
\boxed{\int x^4e^{2x}\,\mathrm{d}x
=e^{2x}\left(\frac12x^4-x^3+\frac32x^2-\frac32x+\frac34\right)+C}
\]
\newpage
\section*{Example 4: \(\int x\ln x\,\mathrm{d}x\)}
Choose the logarithm for the differentiated column:
\[
 u=\ln x,
 \qquad
 \frac{\mathrm{d}v}{\mathrm{d}x}=x
\]
\tableD
From the first diagonal term,
\[
\int x\ln x\,\mathrm{d}x
=\frac{x^2}{2}\ln x-\int \frac{x^2}{2}\cdot\frac{1}{x}\,\mathrm{d}x
\]
Therefore
\[
\int x\ln x\,\mathrm{d}x
=\frac{x^2}{2}\ln x-\frac12\int x\,\mathrm{d}x
=\frac{x^2}{2}\ln x-\frac{x^2}{4}+C
\]
So
\[
\boxed{\int x\ln x\,\mathrm{d}x=\frac{x^2}{2}\ln x-\frac{x^2}{4}+C}
\]

\section*{Example 5: \(\int e^x\sin x\,\mathrm{d}x\)}
Take
\[
 u=\sin x,
 \qquad
 \frac{\mathrm{d}v}{\mathrm{d}x}=e^x
\]
\tableE
Let
\[
I=\int e^x\sin x\,\mathrm{d}x
\]
The table gives
\[
I=e^x\sin x-e^x\cos x+\int (-\sin x)e^x\,\mathrm{d}x
\]
Since
\[
\int (-\sin x)e^x\,\mathrm{d}x=-I
\]
we obtain
\[
I=e^x\sin x-e^x\cos x-I
\]
Hence
\[
2I=e^x(\sin x-\cos x)
\qquad
I=\frac12e^x(\sin x-\cos x)+C
\]
So
\[
\boxed{\int e^x\sin x\,\mathrm{d}x=\frac12e^x(\sin x-\cos x)+C}
\]
